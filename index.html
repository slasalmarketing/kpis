function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Employees"); // 👈 لازم الشيت يكون اسمه كده أو تغيره للاسم اللي عندك if (!sheet) { return ContentService.createTextOutput( JSON.stringify({ error: "Sheet not found" }) ).setMimeType(ContentService.MimeType.JSON); } var lastRow = sheet.getLastRow(); if (lastRow < 2) { return ContentService.createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON); } // جلب الأسماء من العمود A ابتداء من الصف 2 var data = sheet.getRange(2, 1, lastRow - 1, 1).getValues(); var employees = data.map(function (row) { return row[0]; }); return ContentService .createTextOutput(JSON.stringify(employees)) .setMimeType(ContentService.MimeType.JSON); }
